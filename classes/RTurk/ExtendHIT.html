    def to_params
      if assignments.nil? && seconds.nil?
        raise MissingParameters, 'Must add to either the HIT assignment count or expiration time.'
      end
      
      {'HITId' => self.hit_id,
       'MaxAssignmentsIncrement' => self.assignments,
       'ExpirationIncrementInSeconds' => self.seconds}
    end
